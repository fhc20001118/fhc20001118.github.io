回溯算法能解决的问题：

  • 组合问题：N个数里面按一定规则找出k个数的集合
  • 排列问题：N个数按一定规则全排列，有几种排列方式
  • 切割问题：一个字符串按一定规则有几种切割方式
  • 子集问题：一个N个数的集合里有多少符合条件的子集
  • 棋盘问题：N皇后，解数独等等

1、组合总和

难点：1）普通的结束条件是当track中所有数的和等于target时就把track加入到结果数组中，如何优化？

​ 2）如果使用正常的回溯算法会得到结果[2,2,3],[2,3,2],[3,2,2]这样重复的答案，如何去重？

优化1：

每次从选择列表中选出一个数，就用target减去这个数对应的值，这样就不必计算track的和，结束条件就变为 target == 0；非法条件就变为 target < 0。

优化2：

​ 产生重复的原因：在每一个结点做减法展开分支的时候，使用了之前也使用过的数字

​ 解决办法：每一次搜索的时候设置下一次搜索的起点，不要每次从头开始遍历

代码实现：

vector<vector<int>>res;
vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
    vector<int>track;//路径
    backtrack(track, candidates, 0, target);
    return res;
}
void backtrack(vector<int>& track, vector<int>& candidates, int begin, int target) {
    //结束条件，如果遍历到叶节点 0
    if (target == 0) {
        res.push_back(track);
        return;
    }

    //遍历决策树
    for (int i = begin; i < candidates.size(); i++) {
        //因为可以有重复，所以不用判断是否使用了
        //跳过非法条件
        if (target < 0) continue;

        //做出选择
        track.push_back(candidates[i]);
        int k = target - candidates[i];

        //进入下一层决策树
        //因为可以使用重复元素，所以下一次搜索的起点可以从自己开始，所以还是i
        backtrack(track, candidates, i, k);

        //撤销选择
        track.pop_back();
    }
}

2、组合总和II

难点：与I的区别是，每个数字只能使用一次，但是每个数字有多个相同的，如何去除重复的答案？

解决：

​ 先把candidates中的数字进行排序，这样相同的数就会相邻；

​ 然后跳过对同一结点下相等的分支的搜索（比如在结点4中，跳过了对后面两个2的搜索）；

（后面为2的分支得到的结果只会比第一个2的少，并且是它的真子集，所以没有必要）

​

代码实现：

sort(candidates.begin(), candidates.end());//排序
//剪枝优化(关键代码)
void backtrack(vector<int>& track, vector<int>& candidates, int begin, int target) {
        //结束条件
        if (target == 0) {
            res.push_back(track);
            return;
        }

        for (int i = begin; i < candidates.size(); i++) {
            //跳过非法条件
            if (target < 0 ) continue;
            //i > begin的意思是下一个分支，直到找到下一个分支与相邻的分支不等的时候
            if (i > begin && candidates[i] == candidates[i-1]) continue;

            //做出选择
            track.push_back(candidates[i]);

            //从i+1开始是因为自己只能使用一次
            //进入下一层决策树
            backtrack(track, candidates, i+1, target-candidates[i]);

            //撤销选择
            track.pop_back();
        }
}

3、全排列II

难点：如何去重？

解决：

​ 全排列和组合的区别是，全排列每次要重头遍历整个选择列表；

​ 去除相同的数字导致的重复的关键是把nums排序；

​ 但是为了避免出现选择了刚刚选过的数字，用visited数组来标记；

代码块：

​ 

void backtrack(vector<int>& track, vector<int>& nums, vector<int>& visited) {
        //满足结束条件，返回
        if (track.size() == nums.size()) {
            res.push_back(track);
            return;
        }

        //遍历选择列表
        for (int i = 0; i < nums.size(); i++) {
            //跳过非法条件
            if (visited[i] == 1) continue; //如果当前数字已经使用过了就跳过
            //如果当前分支和前一个分支相同，并且前一个分支还未使用的时候跳过
            //其实就是选择了重复元素的第一个
            if (i > 0 && nums[i] == nums[i-1] && visited[i-1] == 0) continue;
            
            //做出选择
            visited[i] = 1;
            track.push_back(nums[i]);

            //进入下一层决策树
            backtrack(track, nums, visited);

            //撤销选择
            visited[i] = 0;
            track.pop_back();
        }
}

3、递增子序列

​ 两个地方需要剪枝：

​ 1、当前选择的结点比父节点小（也就是路径中最后的值）

​ 2、同一层中有数字相等的分支（不管相不相邻）

void backtrack(vector<int>& track, vector<int>& nums, int begin) {
        //这里是判断是否到叶子结点的结束条件，但是有的结果不是叶子结点，所以不需要结束条件
        unordered_set<int>brothers;
        //遍历决策树
        for (int i = begin; i < nums.size(); i++) {
            //跳过非法条件
            if (!track.empty() && nums[i] < track.back()) continue;
            //同一层出现数字相等的分支（不管相不相邻）
            if (brothers.count(nums[i])) continue;

            //做出选择
            track.push_back(nums[i]);
            brothers.insert(nums[i]);
            if (track.size() >= 2) res.push_back(track);

            //进入下一层决策树
            backtrack(track, nums, i+1);

            //撤销选择
            //这里为什么不用brothers.erase(nums[i])?
            //因为brothers存放的是这一层使用过的结点，这个撤销只是说他用完了，而不是没用过
            track.pop_back();
        }
}

5、分割回文串

​ 1）截取子串的决策树怎么画？

​ 切割问题也可以抽象为一颗决策树

​ 解释：

​ track保存的是路径上的信息

void backtrack(vector<string>& track, const string& s, int begin) {
        //结束条件:当截取到字符串末尾时
        if (begin == s.size()) {
            res.push_back(track);
            return;
        }

        //遍历决策树
        for (int i = begin; i < s.size(); i++) {
            //跳过非法条件:不是回文
            if (!isPalindrome(s, begin, i)) continue;

            //做出选择
            string str = s.substr(begin, i-begin+1);//s.substr(pos, len) pos指起始位置，len指截取的长度
            track.push_back(str);

            //进入下一层决策树
            backtrack(track, s, i+1);

            //撤销选择
            track.pop_back();
        }

    }

6、复原ip地址

类似于分割字符串，但是这里有很多细节要处理好

注意：

​ 1、可以直接写一个判断当前分割的子串是否合法的函数，不用在回溯函数里一个个判断，而且到达结束条件的时候还要判断最后一个子字符串是否合法，才能把它加入结果数组

​ 2、记住怎么把字符串变成整型

​ 3、插入点而不是插入字符串，这样操作很方便，撤销的时候也是把刚刚在那个地方插入的点撤销

​ 4、没有使用track保存路径，而是直接在字符串上进行操作，因为返回的是一个一维的字符串数组

bool isValid(const string& s, int start, int end) {
        //到达字符串末尾了
        if (start > end) return false;
        if (s[start] == '0' && start != end) return false;

        int num = 0;
        for (int i = start; i <= end; i++) {
            if (s[i] > '9' || s[i] < '0') return false;
            num = num * 10 + (s[i] - '0');
            if (num > 255) return false;
        }

        return true;
    }
    void backtrack(string& s, int begin, int pointNum) {
        //结束条件：当有三个点时，分割结束
        if (pointNum == 3) {
            //判断第四段子串是否合法
            if (isValid(s, begin, s.size()-1)) {
                res.push_back(s);
                return;
            }
        }
        
        //做出选择
        for (int i = begin; i < s.size(); i++) {
            //跳过非法条件：当前子串是否合法
            if (!isValid(s, begin, i)) continue;
            //做出选择，插入点
            s.insert(s.begin()+i+1, '.');

            //进入下一层决策树
            //注意是i+2，因为插入了一个点
            backtrack(s, i+2, pointNum+1);

            //撤销选择，删除点
            s.erase(s.begin()+i+1);
        }
    }

7、重新安排行程

代码随想录

决策树也比较难想得到，要是能画出决策树，也不难

此部分内容代码随想录讲的很好

vector<string> findItinerary(vector<vector<string>>& tickets) {
        vector<string>track;
        //初始化航班的映射，制作选择列表
        for (const vector<string>& vec : tickets) {
            targets[vec[0]][vec[1]]++;
        }
        //初始化
        track.push_back("JFK");
        backtrack(track, tickets);
        return res;
    }
    bool backtrack(vector<string>& track, const vector<vector<string>>& tickets) {
        //结束条件：如果结果数组的长度等于航班+1
        if (track.size() == tickets.size()+1) {
            res = track; 
            return true;
        }

        for (pair<const string, int>& target : targets[track[track.size()-1]]) {
            //跳过非法条件，可以起飞的次数为0
            //if (target.second <= 0) continue;
            if (target.second > 0) {
                //做出选择
                //cout << target.first << endl;;
                track.push_back(target.first);
                target.second--;

                if (backtrack(track, tickets)) return true; //这样就会保留第一条满足条件的路径，当然也是正确答案

                //撤销选择
                track.pop_back();
                target.second++;
            }
        }
        return false;
    }

注意这里题目条件是按字典排序，因此用map拍好了选择的序列，所以第一条正确的路径即为答案，因此我们可以用bool型作为返回类型

8、解数独

这道题结合了“N皇后” 和 “重新安排行程” 两个问题

选择列表就是1~9这9个数字，结果数组就是棋盘board

这道题的决策树也比较难，相同层的分支是同一个位置的不同选择，不同层的分支是下一个位置的不同选择，所以进入下一层决策树就是在下一个位置进行选择，因此相同层的for循环遍历应该是双循环，从上往下、从左往右遍历整个棋盘，每个空位置选择完毕就进入下一个空位置（也就是下一层决策树）进行选择

在每个空位进行选择数字的时候要判断合法性

​ 1、同一行不能重复

​ 2、同一列不能重复

​ 3、同一个九宫格不能重复（如何判断是否在同一个九宫格也是个难点，具体看代码注释）

bool isValid(int row, int col, char val, const vector<vector<char>>& board) {
        //同一行不能重复
        for (int i = 0; i < 9; i++) {
            if (board[row][i] == val) return false;
        }
        //同一列不能重复
        for (int i = 0; i < 9; i++) {
            if (board[i][col] == val) return false;
        }
        //同一个九宫格不能重复
        //除代表是第几个九宫格，乘代表一个九宫格长度为3，比如行在第2个九宫格，那么前面的长度就是6，所以行从6开始
        int startRow = (row / 3) * 3; 
        int startCol = (col / 3) * 3;

        for (int i = startRow; i < startRow+3; i++) {
            for (int j =startCol; j < startCol+3; j++) {
                if (board[i][j] == val) return false;
            }
        }
        
        return true;
    }
